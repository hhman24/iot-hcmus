#include "WiFi.h"
#include "PubSubClient.h"

const char* ssid = "Wokwi-GUEST";
const char* password = "";
const char* mqttServer = "test.mosquitto.org";
int port = 1883;

WiFiClient espClient;
PubSubClient client(espClient);

void wifiConnect() {
  WiFi.begin(ssid, password);
  while(WiFi.status() != WL_CONNECTED){
    delay(500);
    Serial.print(".");
  }
  Serial.println(" Connected!");
}

void mqttReconnect() {
  while(!client.connected()){
    Serial.print("Attempting MQTT connection...");
    if(client.connect("12345678")){
      Serial.println(" connected");
      client.subscribe("20127102/mssv");
    }else{
      Serial.println(" try again in 5 seconds");
      delay(5000);
    }
  }
}

void callback(char* topic, byte*message, unsigned int length){
  Serial.println(topic);
  String stMessage;
  for(int i = 0; i < length; i++){
    stMessage += (char)message[i];
  }
  Serial.println(stMessage);
if (stMessage == "20127102")
  {
    char buffer[50];
    sprintf(buffer,"Hoàng Hữu Minh An");
    client.publish("20127102/name", buffer);
  }
  else if (stMessage == "20127000"){
    char buffer[50];
    sprintf(buffer,"Trọng Đạt");
    client.publish("20127102/name", buffer);
  }
  else if (stMessage == "20127001"){
    char buffer[50];
    sprintf(buffer,"Hoàng Huỳnh");
    client.publish("20127102/name", buffer);
  }
}

void setup() {
  Serial.begin(9600);
  Serial.print("Connecting to WiFi");

  wifiConnect();

  client.setServer(mqttServer, port);
  client.setCallback(callback);
}

void loop() {
  if(!client.connected()){
    mqttReconnect();
  }
  client.loop();
  delay(1000);
}
